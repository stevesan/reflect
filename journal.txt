- crit damping:
//----------------------------------------
//  Returns the next position for an object being forced by a critically-damped spring attached at pos=0
//	Larger natural-frequency means a faster pull
//----------------------------------------
next_pos = critical_damping( prev_pos, curr_pos, timestep, natural_freq )
A = curr_pos;
curr_speed = (curr_pos - prev_pos) / timestep;
B = curr_speed + natural_freq * curr_pos;
t = timestep;
next_pos = (A+B*t)*Mathf.Exp( -natural_freq*t );
return next_pos;
